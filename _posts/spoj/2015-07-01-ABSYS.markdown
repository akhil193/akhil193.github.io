#include <stdio.h>
#include <stdlib.h>

#define DEBUG 0

int readline ( char * str ) {

	int i = 0;
	char ch;
	while ( (ch=getchar()) != '\n' ) {
		str[i++] = ch;
	}
	str[i] = '\0';
	return i;
}

int is_int ( char * s ) {
	int i = 0;
	int num = 0;

	for ( i = 0 ; s[i] ; i++ ) {
		if ( (s[i] >= '0' && s[i] <= '9') ) {
			num = num * 10 + (s[i] - '0');
		}
		else {
			return -1;
		}
	}
	return num;
}

	int t , z;

}
