void readline( char * a , char * b , int *la , int *lb ){
	
	*la = 0;
	*lb = 0;

	char ch;

	while ( (ch = getchar()) != ' ' ) {
		a[(*la)++] = ch;
	}
	a[*la] = '\0';

	while ( (ch = getchar()) != '\n' ) {
		b[(*lb)++] = ch;
	}
	b[*lb] = '\0';

}
