Following is the solution for Spoj Question: [TEST](http://www.spoj.com/problems/TEST/)

{% highlight cpp %}
#include <cstdio>
#include <iostream>

using namespace std;

int main( int argc, char * argv[] ) {
	int x;
	while(1) {
		scanf("%d", &x);
		if( x == 42 ) {
			break;
		}
		else {
			printf("%d\n", x);
		}
	}
	return 0;
}
