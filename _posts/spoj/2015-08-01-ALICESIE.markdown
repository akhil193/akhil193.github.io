Following is the solution for Spoj Question: [ALICESIE](http://www.spoj.com/problems/ALICESIE/)

{% highlight cpp %}
#include <stdio.h>

int main( int argc , char * argv[] ) {

	int t;
	scanf("%d" , &t);

	while ( t-- ) {

		int n;
		scanf("%d" , &n);
		printf("%d\n" , (n+1)/2);

	}

	return 0;
}
