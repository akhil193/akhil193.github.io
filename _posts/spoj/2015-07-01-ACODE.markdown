#include <stdio.h>
#include <stdlib.h>

int readline ( char * str ) {
	int i = 0;
	char ch = 0;
	while ( (ch = getchar() ) != '\n' ) {
		str[i] = ch;
		i++;
	}
	str[i] = '\0';
	return i;
}

int main ( int argc , char * argv[] ) {

	int len = 0 , i = 0 ;
	char * str = (char *) calloc ( 5010 , sizeof(char) );
	long long int * array = (long long int *) calloc ( 5010 , sizeof(long long int) );

	char string[3] ;

	len = readline ( str );

	while ( str[0] != '0' ) {

}
