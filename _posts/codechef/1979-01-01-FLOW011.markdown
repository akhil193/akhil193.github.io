    while( t-- ) {
        int s;
        scanf("%d", &s);

        float g, h, d;
        if(s < 1500) {
            h = 0.1 * s;
            d = 0.9 * s;
        }
        else {
            h = 500;
            d = 0.98 * s;

    }
