#define ll long long

#define MIN(a, b) a < b ? a : b
#define MAX(a, b) a > b ? a : b

using namespace std;

int main(int argc, char *argv[]) {
    int t;
    scanf("%d", &t);

    while( t-- ) {
        ll int a, b;
        scanf("%lld%lld", &a, &b);
        printf("%lld %lld\n", MAX(a,b), a+b);
    }
    return 0;
}
