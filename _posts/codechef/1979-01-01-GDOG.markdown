#define ll long long

#define MIN(a, b) a < b ? a : b
#define MAX(a, b) a > b ? a : b

using namespace std;

int main(int argc, char *argv[]) {
    int t;
    scanf("%d", &t);

    while( t-- ) {
        int a, b, m = -1;
        scanf("%d%d", &a, &b);
        for(int k = 1; k <= b; k++) {
            m = MAX(m, a%k);
        }
        printf("%d\n", m);
    }
    return 0;
}
