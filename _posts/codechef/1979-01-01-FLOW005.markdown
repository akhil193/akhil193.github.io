using namespace std;

int main(int argc, char *argv[]) {
    int t;
    scanf("%d", &t);
    int array[6] = {100, 50, 10, 5, 2, 1};
    while( t-- ) {
        int n, answer = 0;
        scanf("%d", &n);

        for(int i = 0; i < 6; i++) {
            answer += (n/array[i]);
            n = n % array[i];
        }

        printf("%d\n", answer);
    }
    return 0;
}
